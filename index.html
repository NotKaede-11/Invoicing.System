        from pyodide.ffi import create_proxy

        def format_price(price_str):
            # Converts price string (e.g. "1,999.99") to float by removing commas
            return float(price_str.replace(',', ''))

        def generate_invoice(event):
            # Prevents default form submission
            event.preventDefault()
            
            # Gets currency symbol based on selection
            currency_symbol = '₱' if Element('currency').element.value == 'PHP' else '$'
            
            # Creates decorative ASCII header for invoice
            ascii_banner = '''
                ╔══════════════════════════════════════════╗
                ║             OFFICIAL INVOICE             ║
                ╚══════════════════════════════════════════╝'''
            
            # Builds invoice text with customer information
            invoice_text = f"""{ascii_banner}
                ------------------------------------------------
                Customer Information:
                Name: {Element('name').element.value}
                Email: {Element('email').element.value}
                Phone: {Element('phone').element.value}
                Date: {Element('date').element.value}

                Product Details:
            """
            
            # Gets all product entries from form
            products_container = Element('products-container').element
            product_entries = products_container.getElementsByClassName('product-entry')
            total = 0  # Tracks total price
            
            # Processes each product entry and adds to invoice
            for i, entry in enumerate(product_entries, 1):
                product_name = Element(f'product{i}').element.value
                price = format_price(Element(f'price{i}').element.value)
                total += price
                invoice_text += f"                {i}. {product_name}: {currency_symbol}{price:,.2f}\n"
            
            # Adds total amount to invoice
            invoice_text += f"""
                ------------------------------------------------
                Total Amount: {currency_symbol}{total:,.2f}
                ------------------------------------------------"""
            
            # Displays final invoice
            result_element = Element('result')
            Element('preview').write(invoice_text)
            result_element.element.style.display = "block"

        def add_product(event):

            new_entry.className = 'product-entry'
            new_entry.innerHTML = f'''
                
                
                
                
            '''
            
            # Adds new entry to form
            products_container.appendChild(new_entry)

        def setup():
            # Sets up event handlers for form submission and add product button
            Element('invoiceForm').element.onsubmit = generate_invoice
            Element('addProductBtn').element.onclick = add_product

        # Initializes the application
        setup()
